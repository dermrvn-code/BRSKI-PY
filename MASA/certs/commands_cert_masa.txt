PRIVATE KEY:
	> openssl genrsa -out MASA_priv.key 2048

PUBLIC KEY:
	> openssl pkey -in MASA_priv.key -pubout -out MASA_pub.key

CSR:
	> openssl req -new -key MASA_priv.key -out MASA.csr -subj "/C=DE/CN=MASA"


CRT:
	> openssl x509 -req -in MASA.csr -CA ../ca/MASA_CA.pem -CAkey ../ca/MASA_CA_priv.key -CAcreateserial -out MASA.crt -days 825 -sha256
