PRIVATE KEY:
	> openssl genrsa -out registrar_priv.key 2048

PUBLIC KEY:
	> openssl pkey -in registrar_priv.key -pubout -out registrar_pub.key

CSR:
	> openssl req -new -key registrar_priv.key -out registrar.csr -subj "/C=DE/CN=registrar"

CRT:
	> openssl x509 -req -in registrar.csr -CA ../ca/registrar_CA.pem -CAkey ../ca/registrar_CA_priv.key -CAcreateserial -out registrar.crt -days 825 -sha256
