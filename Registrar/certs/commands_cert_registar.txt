	> openssl genrsa -out registrar_priv.key 2048

CSR:
	> openssl req -new -key registrar_priv.key -out registrar.csr
	DE,NRW,Hamm,HSHL,Trustpoint,Registrar,marvin,grunwald@stud.hshl.de


CRT:
	> openssl x509 -req -in registrar.csr -CA ../../ca/Registrar/registrar_CA.pem -CAkey ../../ca/Registrar/registrar_CA.key -CAcreateserial -out registrar.crt -days 825 -sha256
